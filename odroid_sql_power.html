1. Hardware requirement

1.1 Tp-link HS300

2. Set up mysql if you're first time user

$ sudo mysql -u root # I had to use "sudo" since is new installation

mysql> USE mysql;

mysql> CREATE USER 'YOUR_SYSTEM_USER'@'localhost' IDENTIFIED BY '';

mysql> ALTER USER 'user-name' @ 'localhost' IDENTIFIED BY '**'; //change password

mysql> GRANT ALL PRIVILEGES ON *.* TO 'YOUR_SYSTEM_USER'@'localhost';

mysql> UPDATE user SET plugin='mysql_native_password' WHERE User='YOUR_SYSTEM_USER';

mysql> FLUSH PRIVILEGES;

mysql> exit;

$ service mysql restart

mysql> CREATE DATABASE Power;
